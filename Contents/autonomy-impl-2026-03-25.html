:: PIDロック確認 — 二重起動防止
if exist "%PID_FILE%" (
  tasklist /FI "PID eq %OLD_PID%" | find "%OLD_PID%" >nul
  if not errorlevel 1 (
    echo セッション稼働中: PID %OLD_PID% && exit /b 0
  )
)
claude --continue --channel plugin:discord:discord
