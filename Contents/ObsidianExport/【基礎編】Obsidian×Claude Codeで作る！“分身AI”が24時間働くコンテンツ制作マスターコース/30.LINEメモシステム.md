# LINE×Obsidian オフライン自動整理システム

---

## 0. 目的と条件

* 目的: LINEで書いたメモをVault内で自動分類し所定フォルダへ振り分ける。API不要。完全オフライン。
* 対象: 初学者。Windows / macOS / Linux。Python 3.10+ 推奨。
* 内部パスはASCIIのみ。表示名はフロントマターに保持。
* GUIは任意。CLIを既定。GUI不可時は自動でCLIにフォールバック。

---

## 2. LINEからの取り込み方法（最低1方式を明記）

* 方式A: 手動エクスポート
  PC版LINEでメモをコピー→`LINE_Inbox/`に`.md`で保存。既定の監視はポーリングで自動処理。
* 方式B: モバイル共有→同期フォルダ
  iOS/Androidで「共有」→「ファイルに保存」→クラウド同期フォルダ`LINE_Inbox/`を指定。PC側で処理。
* 方式C: 自動貼り付け用ショートカット
  OSショートカットでクリップボード内容を`LINE_Inbox/`に新規`.md`保存（任意）。

## 4. CLAUDE.md（20行厳守・そのまま配布）

```markdown
目的: LINE_Inbox の未処理Markdownを5分類し移動する。
制約1: 完全オフライン。外部APIとネットワーク禁止。
制約2: ユーザー本文は一切改変しない。
制約3: 破壊的操作禁止。バックアップとログ必須。
開始条件: tools/rules/triggers.yaml のキーワードに一致。
参照ルール: tools/rules/classification.yaml と tools/rules/naming.yaml。
対象: Vault/LINE_Inbox/*.md の未処理ファイル。
未処理判定: フロントマター processed: false または欠落。
命名: naming.yaml の規則でファイル名を生成。
タグ: 日本語主体 固有名詞は英語 上限5個。
分類: 優先度順で最初の一致へ移動。複数一致は優先度で解決。

冪等性: 処理後に processed: true を書き込む。
ログ: tools/system.log と tools/error.log に追記。
フォルダ: 不足時は自動作成。安全なパス結合を使用。
実行1: python3 tools/line_sorter_offline.py --once
実行2: 連続監視は --watch --interval 10
テスト: python3 tools/test_system.py を実行し合格を確認。
禁止: 本ファイルの編集 推測実行 依存の追加。
出力: 標準出力でサマリー GUIあれば同時反映。
```

## 6. 整理エンジン（Python・標準実行はCLI）

`tools/line_sorter_offline.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import argparse, re, sys, time, shutil, logging
from logging.handlers import RotatingFileHandler
from pathlib import Path
from datetime import datetime

VAULT = Path(__file__).resolve().parents[1]
TOOLS = VAULT / "tools"
RULES = TOOLS / "rules"
INBOX_LINE = VAULT / "LINE_Inbox"
FALLBACK = VAULT / "Inbox"
CATS = [VAULT/"01_Ideas", VAULT/"02_Materials", VAULT/"03_Contents"]
LOG_SYS = TOOLS / "system.log"
LOG_ERR = TOOLS / "error.log"

def setup_logging():
    TOOLS.mkdir(parents=True, exist_ok=True)
    for p in [LOG_SYS, LOG_ERR]:
        p.parent.mkdir(parents=True, exist_ok=True)
        if not p.exists(): p.touch()
    h1 = RotatingFileHandler(LOG_SYS, maxBytes=1_000_000, backupCount=3, encoding="utf-8")
    h2 = RotatingFileHandler(LOG_ERR, maxBytes=1_000_000, backupCount=3, encoding="utf-8")
    logging.basicConfig(level=logging.INFO, handlers=[h1])
    err = logging.getLogger("err")
    err.setLevel(logging.ERROR)
    err.addHandler(h2)
    return logging.getLogger(__name__), err

def require_yaml():
    try:
        import yaml  # type: ignore
        return yaml
    except Exception:
        print("PyYAMLが見つかりません。まず: python3 tools/install.py を実行してください。", file=sys.stderr)
        sys.exit(2)

def load_rules(yaml):
    with open(RULES/"naming.yaml", "r", encoding="utf-8") as f:
        naming = yaml.safe_load(f) or {}
    with open(RULES/"classification.yaml", "r", encoding="utf-8") as f:
        cls = yaml.safe_load(f) or {}
    priority = cls.get("priority", [])
    rules = cls.get("rules", {})
    fallback = cls.get("fallback", {"path": "Inbox"})["path"]
    return naming, priority, rules, fallback

def ensure_dirs():
    (VAULT/"LINE_Inbox").mkdir(exist_ok=True)
    (VAULT/"Inbox").mkdir(exist_ok=True)
    for d in CATS: d.mkdir(exist_ok=True)

FRONTMATTER_RE = re.compile(r"^---\n(.*?)\n---\n", re.S)
def parse_frontmatter(text, yaml):
    m = FRONTMATTER_RE.match(text)
    if not m: return {}, text
    fm_text = m.group(1)
    fm = yaml.safe_load(fm_text) or {}
    body = text[m.end():]
    return fm, body

def compose_frontmatter(d, yaml):
    dumped = yaml.safe_dump(d, allow_unicode=True, sort_keys=False).strip()
    return f"---\n{dumped}\n---\n"

INVALID = re.compile(r'[\\/:*?"<>|]+')
def sanitize_title(s, max_len=40):
    s = INVALID.sub("_", s).strip()
    return s[:max_len] if len(s) > max_len else s

def infer_title(body):
    for line in body.splitlines():
        if line.strip().startswith("# "):
            return line.strip()[2:].strip()
    for line in body.splitlines():
        t = line.strip()
        if t: return t[:40]
    return "Untitled"

def build_tags(text, limit=5):
    # 極小ロジック: 先頭#タグ行 or キーワード抽出の簡易版
    tags = []
    for w in ["Udemy","YouTube","X","Instagram","Kindle","Podcast","Newsletter","アイデア","資料","コンテンツ"]:
        if w in text and len(tags) < limit: tags.append(w)
    return tags

def classify(title, body, priority, rules):
    target = f"{title}\n{body}"
    for cat in priority:
        spec = rules.get(cat, {})
        kws = spec.get("keywords", []) or []
        if any(k in target for k in kws):
            return spec.get("path", "Inbox")
    return "Inbox"

def unique_path(base: Path) -> Path:
    if not base.exists(): return base
    stem, suf = base.stem, base.suffix
    for i in range(1, 1000):
        cand = base.with_name(f"{stem}_{i}{suf}")
        if not cand.exists(): return cand
    return base.with_name(f"{stem}_{int(time.time())}{suf}")

def process_once(args):
    yaml = require_yaml()
    log, err = setup_logging()
    ensure_dirs()
    naming, priority, rules, fallback = load_rules(yaml)
    fmt = naming.get("timestamp_format", "%Y%m%d_%H%M")
    tag_limit = naming.get("tag_limit", 5)
    processed_key = naming.get("frontmatter_keys",{}).get("processed","processed")
    title_key = naming.get("frontmatter_keys",{}).get("title","title")
    tags_key = naming.get("frontmatter_keys",{}).get("tags","tags")

    files = sorted(INBOX_LINE.glob("*.md"))
    moved, errors, skipped = 0, 0, 0
    for src in files:
        try:
            text = src.read_text(encoding="utf-8")
            fm, body = parse_frontmatter(text, yaml)
            if fm.get(processed_key, False) is True:
                skipped += 1
                continue
            title = fm.get(title_key) or infer_title(body)
            dest_dir_name = classify(title, body, priority, rules)
            dest_dir = (VAULT / dest_dir_name) if dest_dir_name else FALLBACK
            if not dest_dir.exists(): dest_dir.mkdir(parents=True, exist_ok=True)

            ts = datetime.now().strftime(fmt)
            new_title = sanitize_title(title, naming.get("max_title_len", 40))
            dest = unique_path(dest_dir / f"{ts}_{new_title}.md")

            tags = fm.get(tags_key) or build_tags(f"{title}\n{body}", tag_limit)
            new_fm = dict(fm)
            new_fm[processed_key] = True
            new_fm[title_key] = title
            new_fm[tags_key] = tags

            out = compose_frontmatter(new_fm, yaml) + body
            if args.dry_run:
                log.info(f"[DRY] {src.name} -> {dest}")
            else:
                dest.write_text(out, encoding="utf-8")
                src.unlink()
            moved += 1
        except Exception as e:
            err.error(f"{src.name}: {e}")
            # フォールバック退避
            try:
                fb = unique_path(FALLBACK / src.name)
                if not args.dry_run:
                    shutil.move(str(src), str(fb))
            except Exception as e2:
                err.error(f"fallback failed {src.name}: {e2}")
            errors += 1
    print(f"done moved={moved} skipped={skipped} errors={errors}")

def watch_loop(interval, args):
    while True:
        process_once(args)
        time.sleep(interval)

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--once", action="store_true", help="一回だけ実行")
    ap.add_argument("--watch", action="store_true", help="連続監視")
    ap.add_argument("--interval", type=int, default=10, help="監視間隔秒")
    ap.add_argument("--dry-run", action="store_true")
    args = ap.parse_args()

    if not (args.once or args.watch):
        args.once = True
    if args.once:
        process_once(args)
    else:
        watch_loop(args.interval, args)

if __name__ == "__main__":
    main()
```

## 8. オフライン依存の導入

`tools/install.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import subprocess, sys, os
from pathlib import Path
import venv

ROOT = Path(__file__).resolve().parents[1]
TOOLS = ROOT / "tools"
VENV = TOOLS / ".venv"
WHEEL = TOOLS / "vendor" / "pyyaml-6.0.1-py3-none-any.whl"

def main():
    print("creating venv:", VENV)
    venv.EnvBuilder(with_pip=True).create(VENV)
    py = VENV / ("Scripts/python.exe" if os.name == "nt" else "bin/python")
    if not WHEEL.exists():
        print("vendorにPyYAMLのwhlがありません。配置後に再実行してください。")
        print("期待ファイル:", WHEEL)
        sys.exit(2)
    print("installing PyYAML from local wheel")
    subprocess.check_call([str(py), "-m", "pip", "install", "--no-index", "--find-links", str(WHEEL.parent), "pyyaml"])
    print("OK. 実行例:")
    if os.name == "nt":
        print(rf"{VENV}\Scripts\python tools\line_sorter_offline.py --once")
    else:
        print(f"{py} tools/line_sorter_offline.py --once")
    print("環境変数: PYTHONUTF8=1 を推奨")

if __name__ == "__main__":
    main()
```

## 11. 運用ルール

* タグは最大5個。増やす場合も10個まで。
* 誤分類は`classification.yaml`にキーワードを追加して調整。
* フォルダは自動生成。失敗時は`Inbox/`に退避しログに記録。
* ログ: `tools/system.log` と `tools/error.log`。

---

## 12. トラブルシューティング

* 文字化け: すべてUTF-8で保存。Windowsは`set PYTHONUTF8=1`。
* PyYAML未導入: `python3 tools/install.py` を実行。`tools/vendor/`にwhlを同梱。
* GUIが起動しない: `python3 tools/line_sorter_offline.py --once` を使用。
* ファイルが動かない: `LINE_Inbox/`に`.md`を置いてから実行。
